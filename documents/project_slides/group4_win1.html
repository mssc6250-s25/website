    poly_mae = mean_absolute_error(y_test, y_poly_pred)
    poly_r2 = r2_score(y_test, y_poly_pred)

    poly_results[degree] = {
        "MSE": poly_mse,
        "MAE": poly_mae,
        "R2": poly_r2
    }

    print(f" Degree {degree} - MSE: {poly_mse:.4f}, MAE: {poly_mae:.4f}, R²: {poly_r2:.4f}")
